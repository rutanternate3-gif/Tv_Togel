  <script>
    // ------------------------------
    // Mock data + client fetch logic
    // ------------------------------

    // NOTE: Untuk produksi, ganti `DATA_API` ke endpoint yang mengembalikan JSON
    // Format contoh yang diharapkan (array hasil):
    /*
    [
      {"market":"Singapore","period":"2025-11-22","number":"1234","note":"Malem"},
      {"market":"Hongkong","period":"2025-11-22","number":"5678","note":"Siang"}
    ]
    */

    const DATA_API = null; // <-- ganti dengan URL API Anda, misal: 'https://api.example.com/results'

    const mock = {
      lastUpdate: new Date().toLocaleString('id-ID'),
      results: [
        {market: 'Singapore', period: '2025-11-22 17:45', number: '6321', note: 'Malam'},
        {market: 'Hongkong', period: '2025-11-22 16:00', number: '0842', note: 'Siang'},
        {market: 'Sydney', period: '2025-11-22 15:00', number: '7139', note: 'Sore'}
      ],
      announcements: 'Server maintenance setiap Senin 02:00-03:00 WIB. Gunakan link alternatif saat gangguan.',
      altLinks: [
        {text: 'Link 1', href: '#'},
        {text: 'Link 2', href: '#'}
      ],
      stats: {totalMarkets: 12, lastFetched: '3 menit lalu'}
    }

    function renderResults(data){
      const tbody = document.querySelector('#resultsTable tbody');
      tbody.innerHTML = '';
      data.results.forEach(r=>{
        const tr = document.createElement('tr');
        tr.innerHTML = `<td>${escapeHtml(r.market)}</td><td>${escapeHtml(r.period)}</td><td class='num'>${escapeHtml(r.number)}</td><td class='small'>${escapeHtml(r.note||'')}</td>`;
        tbody.appendChild(tr);
      });
      document.getElementById('lastUpdate').textContent = data.lastUpdate || '-';
      document.getElementById('announcements').textContent = data.announcements || '-';
      const alt = document.getElementById('altLinks'); alt.innerHTML = '';
      (data.altLinks||[]).forEach(a=>{ const el=document.createElement('a'); el.href=a.href; el.textContent=a.text; alt.appendChild(el)});
      document.getElementById('stats').textContent = `Pasaran: ${data.stats?.totalMarkets || '-'} • Terakhir: ${data.stats?.lastFetched || '-'}`;
    }

    function escapeHtml(s){ if(!s && s!==0) return ''; return String(s).replace(/[&<>"']/g, function(c){ return {'&':'&amp;','<':'&lt;','>':'&gt;','"':'&quot;',"'":"&#39;"}[c]; }); }

    async function fetchData(){
      try{
        if(!DATA_API){
          // gunakan data mock
          renderResults(mock);
          return;
        }
        const resp = await fetch(DATA_API);
        if(!resp.ok) throw new Error('Fetch error: '+resp.status);
        const data = await resp.json();
        // normalize minimal structure
        const payload = {
          lastUpdate: new Date().toLocaleString('id-ID'),
          results: data.results || data || [],
          announcements: data.announcements || '',
          altLinks: data.altLinks || [],
          stats: data.stats || {}
        };
        renderResults(payload);
      }catch(err){
        console.error(err);
        // fallback ke mock saat error
        renderResults(mock);
      }
    }

    document.getElementById('refreshBtn').addEventListener('click', (e)=>{ e.preventDefault(); fetchData(); });

    // auto load
    fetchData();
  </script>
